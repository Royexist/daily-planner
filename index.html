    <div class="header">

        <input type="text" id="taskInput" placeholder="添加新任务...">

            <option value="morning">早上</option>

        <textarea id="reflection" placeholder="记下今天的闪光点或偏差..." oninput="saveData()"></textarea>

<script>

    // 默认初始任务清单

    const defaultTasks = [

        { id: 1, text: "P1 | 完善大纲/小说一章", period: "morning", completed: false },

        { id: 2, text: "P2 | 英语单词 (30min)", period: "morning", completed: false },

        { id: 3, text: "P3 | 外文阅读", period: "morning", completed: false },

        { id: 4, text: "P1 | 法考视频", period: "afternoon", completed: false },

        { id: 5, text: "P2 | 专业教材", period: "afternoon", completed: false },

        { id: 6, text: "复盘今日/复习以前/Rest", period: "evening", completed: false },

        { id: 7, text: "睡前课外书", period: "evening", completed: false }

    ];


    let state = {

        date: new Date().toLocaleDateString(),

        tasks: defaultTasks,

        reflection: ""

    };


    function init() {

        const saved = localStorage.getItem('full_schedule_data');

        if (saved) {

            const parsed = JSON.parse(saved);

            // 如果日期相同，加载数据；如果日期不同（第二天），重置勾选状态但保留任务结构

            if (parsed.date === state.date) {

                state = parsed;

            } else {

                state.tasks = parsed.tasks.map(t => ({ ...t, completed: false }));

                state.reflection = ""; 

            }

        }

        document.getElementById('dateDisplay').innerText = state.date;

        document.getElementById('reflection').value = state.reflection;

        render();

    }


    function render() {

        const pMap = { morning: '早上', afternoon: '下午', evening: '晚上' };

        Object.keys(pMap).forEach(p => {

            const el = document.getElementById(p);

            el.innerHTML = `<div class="section-title">${pMap[p]}</div>`;

            

            state.tasks.filter(t => t.period === p).forEach(task => {

                const div = document.createElement('div');

                div.className = `task-item ${task.completed ? 'completed' : ''}`;

                div.innerHTML = `

                    <input type="checkbox" ${task.completed ? 'checked' : ''} onclick="toggleTask(${task.id})">

                    <span class="task-text" onclick="editTask(${task.id})">${task.text}</span>

                    <button class="btn-del" onclick="deleteTask(${task.id})">删除</button>

                `;

                el.appendChild(div);

            });

        });

        saveData();

    }


    function addTask() {

        const input = document.getElementById('taskInput');

        if (!input.value) return;

        state.tasks.push({

            id: Date.now(),

            text: input.value,

            period: document.getElementById('periodInput').value,

            completed: false

        });

        input.value = '';

        render();

    }


    function toggleTask(id) {

        const task = state.tasks.find(t => t.id === id);

        if(task) task.completed = !task.completed;

        render();

    }


    function deleteTask(id) {

        state.tasks = state.tasks.filter(t => t.id !== id);

        render();

    }


    function editTask(id) {

        const task = state.tasks.find(t => t.id === id);

        const next = prompt("修改任务:", task.text);

        if (next) { task.text = next; render(); }

    }


    function saveData() {

        state.reflection = document.getElementById('reflection').value;

        localStorage.setItem('full_schedule_data', JSON.stringify(state));

    }


    init();

</script>
